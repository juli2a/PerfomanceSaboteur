// store/sidebar-unstable.ts
export const useSidebarStore = create<SidebarState>()(
  persist(
    (set) => ({
      collapsed: false,
      setCollapsed: (collapsed) => set({ collapsed }),
    }),
    { name: "sidebar-collapsed-unstable" },
  ),
);

// Server has no idea what the user picked last time — it always renders
// collapsed=false, and localStorage's real value only lands after mount.
