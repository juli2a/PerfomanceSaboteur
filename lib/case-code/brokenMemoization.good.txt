// `products` never changes while the threshold slider moves, so this
// pipeline runs exactly once instead of on every tick.
const sparklines = useMemo(
  () => products.map((p) => processSparklineHistory(p.rawHistory)),
  [products],
);

{products.map((product, i) => (
  <MicroCard
    key={product.id}
    title={product.meta.title}
    sku={product.meta.sku}
    marginality={product.marginality}
    value={formatCurrency(product.metrics.currentValue)}
    rating={product.metrics.rating}
    sparklineData={sparklines[i]}   // same array reference every render
    lowMargin={product.marginality < threshold}
  />
))}

// components/dashboard/MicroCard.tsx
// title/sku/marginality/rating are primitives off an unchanging record;
// value is freshly formatted but always identical content; sparklineData is
// the same array reference every render. lowMargin is the only prop that
// actually changes, and only for the handful of cards whose marginality
// sits at the current threshold — so memo's comparison skips re-rendering
// everything else.
export default memo(function MicroCard(props: Props) {
  return <MicroCardView {...props} />;
});

// lib/utils/sparkline-processing.ts
export function processSparklineHistory(raw: number[]): number[] {
  const cleaned = removeOutliersIterative(raw);   // IQR trim, up to 5 passes
  const smoothed = movingAverage(cleaned, 7);      // 7-day trailing average
  return downsampleTo(smoothed, 7);                // → the 7 points the sparkline shows
}

// 365 raw points behind each card never change while the user drags the
