// store/inventory-status.ts — plain Zustand, atomic per-product reads
export const useInventoryStatusStore = create<InventoryStatusState>((set) => ({
  statuses: new Map(),
  setStatuses: (ids, status) =>
    set((state) => {
      const next = new Map(state.statuses);
      for (const id of ids) next.set(id, status);
      return { statuses: next };
    }),
}));

// components/inventory/ProductCard.tsx

// Each card subscribes only to `state.statuses.get(product.id)` — Zustand
// re-renders a component only when the *value its own selector returns*
// changes. Changing one product's status updates that one map entry, so
// only the matching card's selector re-evaluates to a different value;
// every other card's selector still returns the same status it had before
// and skips re-rendering entirely.
